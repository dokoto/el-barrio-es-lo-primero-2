3ยบ EL_BARRIO_ES_LO_PRIMERO->TARGET->Build Rules
 - Crear nueva regla con nombre: Copy Resources
 - Meter el siguiente script 

	#!/bin/bash
	FOLDER_LEVEL="../../../../" ;
	IMG_FOLDER="img" ;
	AUDIO_FOLDER="audio" ;
	TTF_FOLDER="ttf" ;
	
	if [ ! -d $FOLDER_LEVEL"$IMG_FOLDER" ]; then
		echo $IMG_FOLDER ", no found" ;
	fi
	
	if [ ! -d $FOLDER_LEVEL"$AUDIO_FOLDER" ]; then
		echo $AUDIO_FOLDER ", no found" ;
	fi
	
	if [ ! -d $FOLDER_LEVEL"$TTF_FOLDER" ]; then
		echo $TTF_FOLDER ", no found" ;
	fi
